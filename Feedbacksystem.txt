  <script>
    (function(){
      const q5 = document.getElementById('q5');
      const q5val = document.getElementById('q5val');
      const form = document.getElementById('feedbackForm');
      const resultBox = document.getElementById('resultBox');
      const resultPre = document.getElementById('resultPre');
      const overlay = document.getElementById('overlay');
      const closeBtn = document.getElementById('closeBtn');

      // Update rating display
      q5.addEventListener('input', function(){
        q5val.textContent = q5.value;
      });

      // Simple form submit handler
      form.addEventListener('submit', function(e){
        e.preventDefault();

        // collect values
        const data = {
          q1: document.getElementById('q1').value.trim(),
          q2: document.getElementById('q2').value,
          q3: document.getElementById('q3').value.trim(),
          q4: document.getElementById('q4').value.trim(),
          q5: Number(document.getElementById('q5').value),
          q6: document.getElementById('q6').value.trim(),
          submittedAt: new Date().toISOString()
        };

        // Basic validation: require at least q2 and q5
        if (!data.q2) {
          alert('Please select how often you use our services (Q2).');
          return;
        }

        // Show captured data
        resultPre.textContent = JSON.stringify(data, null, 2);
        overlay.style.display = 'block';
        resultBox.style.display = 'block';
        resultBox.focus();

        // In a real implementation you'd send `data` with fetch() to your backend, e.g.
        // fetch('/api/feedback', { method:'POST', body: JSON.stringify(data), headers:{'Content-Type':'application/json'} })

        // Optionally reset form after submit:
        // form.reset();
        // q5val.textContent = '7';
      });

      // Close modal
      function closeModal(){
        overlay.style.display = 'none';
        resultBox.style.display = 'none';
      }
      closeBtn.addEventListener('click', closeModal);
      overlay.addEventListener('click', closeModal);
      document.addEventListener('keydown', function(e){
        if (e.key === 'Escape') closeModal();
      });
    })();
  </script>
